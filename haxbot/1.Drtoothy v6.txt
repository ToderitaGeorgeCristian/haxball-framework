function download(data, filename, type) {
  let file = new Blob([data], { type: type });
  let a = document.createElement("a"),
    url = URL.createObjectURL(file);
  a.href = url;
  a.download = filename;
  document.body.appendChild(a);
  a.click();
  setTimeout(function () {
    document.body.removeChild(a);
    window.URL.revokeObjectURL(url);
  }, 0);
}

room.onPlayerJoin = function(player) {
	room.sendAnnouncement(credits, player.id, 0xFF0000, "bold", 0);
	room.sendAnnouncement("Welcome " + player.name + " ! Commands: !help, !stats, !version", player.id, 0xFF0000, "bold", 1);
	
	for (let i = 0; i < room.getPlayerList().length; i++) {
		if ( getAuth(room.getPlayerList()[i].id) == player.auth  ) {
			room.kickPlayer(room.getPlayerList()[i].id, 'Duplicate player kick', false);
		}
	}
	
	setTimeout(function() {
		if ( Object.keys(dbPlayers).includes(player.auth) ) {
		if ( getAuthValue('permBan', true).includes(player.auth) ) {
			room.kickPlayer(player.id, 'You are permanently banned !', true);
		}
		dbPlayers[player.auth].id = player.id;
		dbPlayers[player.auth].name = player.name;
		if ( dbPlayers[player.auth].role != "player" ) {
			room.setPlayerAdmin(player.id, true);
			room.sendChat((dbPlayers[player.auth].role == "master" ? "Master " : "Admin ") + player.name + " has connected to the room !");
		}
	} 
	else {
		dbPlayers[player.auth] = {
			id: player.id,
			name: player.name,
			role: 'player',
			mute: false,
			tempBan: false,
			permBan: false,
			Goals:0,
			Assist:0,
			GamesPlayed:0
		};
		if ( superadmin.includes(player.auth) ) {
			room.setPlayerAdmin(player.id, true);
			dbPlayers[player.auth].role = 'master';
			room.sendAnnouncement(player.name + " is now a room master !", null, 0xFFFF00, "bold", 0);	
		}
	} 
		
	
	},1*1000);
	
	
};

room.onPlayerKicked = function(kickedPlayer, reason, ban, byPlayer) {
	if (kickedPlayer.id) {
		if ( ban ) {
			if ( dbPlayers[getAuth(kickedPlayer.id)].role == "master" ) {
				room.clearBan(kickedPlayer.id);
				dbPlayers[getAuth(byPlayer.id)].role = "player";
				room.kickPlayer(byPlayer.id, 'Action denied !', false);
			} else {
				dbPlayers[getAuth(kickedPlayer.id)].tempBan = true;
			}
		} else {
			if ( dbPlayers[getAuth(kickedPlayer.id)].role == "master" ) {
				room.clearBan(kickedPlayer.id);
				dbPlayers[getAuth(byPlayer.id)].role = "player";
				room.kickPlayer(byPlayer.id, 'Action denied !', false);
			}
		}
	}
};

room.onPlayerAdminChange = function(changedPlayer, byPlayer) {
	if ( byPlayer.id != 0 && dbPlayers[getAuth(changedPlayer.id)].role == "master" ) {

	}
	dbPlayers[getAuth(changedPlayer.id)].mute = false;
}

				room.sendChat("[RS]Only masters can save DB !",player.id);
			}
		}
		else if (["!realmode"].includes(message[0].toLowerCase())){
			if (message.length != 2) {
				room.sendChat("[RS] Comanda incorecta. Vezi !admin.",player.id);
			}
			if (message.length == 2) {
				if ( message[1] == "on") {
					realmode = true;
					room.sendChat("[RS] Real Mode Activated");
				}
				if ( message[1] == "off") {
					realmode = true;
					room.sendChat("[RS] Real Mode DE-Activated");
				}
			}
		}
		else if (["!map"].includes(message[0].toLowerCase())){
			if (message.length != 2) {
				room.sendChat("[RS] Harti: RS6xMaximo, RS6xVulon, RS7xVulon, RS5xMinimo, RSGrandev87, RSRawR12C", player.id);
			}
			if (message.length == 2) {
				if ( message[1] == "RS6xMaximo") {
					fRS6xMaximo();
					return false;
				}
				else if ( message[1] == "RS6xVulon") {
					fRS6xVulon();
					return false;
				}
				else if ( message[1] == "RS7xVulon") {
					fRS7xVulon();
					return false;
				}
				else if ( message[1] == "RS5xMinimo") {
					fRS5xMinimo();
					return false;
				}
				else if ( message[1] == "RSGrandev87") {
					fRSGrandev87();
					return false;
				}
				else if ( message[1] == "RSRawR12C") {
					fRSRawR12C();
					return false;
				}
				else {
					room.sendChat("[RS] Nu exista harta. Vezi !map", player.id);
				}
			}
		}
		else if (["!resetstats"].includes(message[0].toLowerCase())){
			if ( dbPlayers[getAuth(player.id)].role == "master" ) {
				Object.keys(dbPlayers).forEach(key => {
					dbPlayers[key].GamesPlayed = 0;
					dbPlayers[key].Goals = 0;
					dbPlayers[key].Assist = 0
				})
				room.sendChat("[RS] ALL DB Stats reseted !",player.id);
			} else {
				room.sendChat("[RS]Only masters can reset DB stats !",player.id);
			}
		}
	}
	if (message[0][0] == "!") {
		return false;
	}
	if (player.id) {
		if ( getAuth(player.id) && dbPlayers[getAuth(player.id)].mute == true ) {
			room.sendChat("You are muted.", player.id);
			return false;
		}
	}
}

room.onTeamVictory = function(scores) {
for (let i = 0; i < room.getPlayerList().length; i++) {
		if ( room.getPlayerList()[i].team !== 0 ) {
			dbPlayers[getAuth(room.getPlayerList()[i].id)].GamesPlayed++
		}
	}
}
